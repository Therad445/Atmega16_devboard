#include <avr/io.h>
#define F_CPU 8000000UL
#include <util/delay.h>



int main(void)
{
	DDRA = 0x00;
	DDRD = 0xFF;
	PORTA |= (1<<1)|(1<<2)|(1<<3)|(1<<4);
	
	while(1)
	{
		if (!(PINA&0b00000010))
		{
			PORTD |= (1<<6)|(1<<7);
		} else
			{
				PORTD &= ~((1<<6)|(1<<7));
			}
		if (!(PINA&0b00000100))
		{
			PORTD |= (1<<2)|(1<<5);
		}	else
			{
				PORTD &= ~((1<<2)|(1<<5));
			}
		if (!(PINA&0b00001000))
		{
			PORTD |= (1<<1)|(1<<4);
		} else
			{
				PORTD &= ~((1<<1)|(1<<4));
			}
		if (!(PINA&0b00010000))
		{
			PORTD |= (1<<0)|(1<<3);
		} else
			{
				PORTD &= ~((1<<0)|(1<<3));
			}
	}
}
